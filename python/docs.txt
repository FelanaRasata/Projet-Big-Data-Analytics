Step - 1

Ensure you've installed Java and Hadoop

If not read this `https://www.tutorialspoint.com/sqoop/sqoop_installation.htm`

Step - 2

wget https://archive.apache.org/dist/sqoop/1.4.7/sqoop-1.4.7.bin__hadoop-2.6.0.tar.gz

tar -xvzf sqoop-1.4.7.bin__hadoop-2.6.0.tar.gz

export HADOOP_HOME=/usr/local/hadoop
export HADOOP_MAPRED_HOME=$HADOOP_HOME
export HADOOP_COMMON_HOME=$HADOOP_HOME
export HADOOP_HDFS_HOME=$HADOOP_HOME
export YARN_HOME=$HADOOP_HOME
export HADOOP_COMMON_LIB_NATIVE_DIR=$HADOOP_HOME/lib/native
export SQOOP_HOME=/usr/local/sqoop

export PATH=$PATH:$JAVA_HOME/bin:$HADOOP_HOME/sbin:$HADOOP_HOME/bin:$SQOOP_HOME/bin

sqoop import --connect jdbc:mongodb://localhost:27017/sourceCSV --collection catalogues --hive-import --hive-database default --hive-table catalogue_hive --columns marque,nom,puissance,longueur,nbPlaces,nbPortes,couleur,occasion,prix --split-by _id
